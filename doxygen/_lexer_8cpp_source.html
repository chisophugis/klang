79  }
80

82  // Comment until end of line.
83  do LastChar = GetCharFromBuffer();
84  while (LastChar != KLANG_LEXER_EOF && LastChar != '\n' && LastChar != '\r');
85

93  if (LastChar == KLANG_LEXER_EOF) {

95  return;
96  }
97 
98  // Otherwise, just return the character as its ascii value.
99  int ThisChar = LastChar;
100  LastChar = GetCharFromBuffer();
101  Result.Kind = ThisChar;
102  return;
103 }
