33 ExprAST *Parser::ParseIdentifierExpr() {
34  std::string IdName = Tok.IdentifierStr;
35 
36  GetNextToken(); // eat identifier.
37 
38  if (Tok.Kind != '(') // Simple variable ref.
39  return new VariableExprAST(IdName);
40 
41  // Call.
42  GetNextToken(); // eat (
43  std::vector<ExprAST*> Args;
44  if (Tok.Kind != ')') {
45  while (1) {
46  ExprAST *Arg = ParseExpression();
47  if (!Arg) return 0;
48  Args.push_back(Arg);
49 
50  if (Tok.Kind == ')') break;
51 
52  if (Tok.Kind != ',')
53  return Error("Expected ')' or ',' in argument list");
54  GetNextToken();
55  }
56  }
57 
58  // Eat the ')'.
59  GetNextToken();
60 
61  return new CallExprAST(IdName, Args);
62 }

155 ExprAST *Parser::ParseVarExpr() {
156  GetNextToken(); // eat the var.
157 
158  std::vector<std::pair<std::string, ExprAST*> > VarNames;
159 
160  // At least one variable name is required.
161  if (Tok.Kind != tok::tok_identifier)
162  return Error("expected identifier after var");
163 
164  while (1) {
165  std::string Name = Tok.IdentifierStr;
166  GetNextToken(); // eat identifier.
167 
168  // Read the optional initializer.
169  ExprAST *Init = 0;
170  if (Tok.Kind == '=') {
171  GetNextToken(); // eat the '='.
172 
173  Init = ParseExpression();
174  if (Init == 0) return 0;
175  }
176 
177  VarNames.push_back(std::make_pair(Name, Init));
178 
179  // End of var list, exit loop.
180  if (Tok.Kind != ',') break;
181  GetNextToken(); // eat the ','.
182 
183  if (Tok.Kind != tok::tok_identifier)
184  return Error("expected identifier list after var");
185  }
186 
187  // At this point, we have to have 'in'.
188  if (Tok.Kind != tok::tok_in)
189  return Error("expected 'in' keyword after 'var'");
190  GetNextToken(); // eat 'in'.
191 
192  ExprAST *Body = ParseExpression();
193  if (Body == 0) return 0;
194 
195  return new VarExprAST(VarNames, Body);
196 }

238 ExprAST *Parser::ParseBinOpRHS(int ExprPrec, ExprAST *LHS) {
239  // If this is a binop, find its precedence.
240  while (1) {
241  int TokPrec = Tok.GetTokPrecedence();
242 
243  // If this is a binop that binds at least as tightly as the current binop,
244  // consume it, otherwise we are done.
245  if (TokPrec < ExprPrec)
246  return LHS;
247 
248  // Okay, we know this is a binop.
249  int BinOp = Tok.Kind;
250  GetNextToken(); // eat binop
251 
252  // Parse the unary expression after the binary operator.
253  ExprAST *RHS = ParseUnary();
254  if (!RHS) return 0;
255 
256  // If BinOp binds less tightly with RHS than the operator after RHS, let
257  // the pending operator take RHS as its LHS.
258  int NextPrec = Tok.GetTokPrecedence();
259  if (TokPrec < NextPrec) {
260  RHS = ParseBinOpRHS(TokPrec+1, RHS);
261  if (RHS == 0) return 0;
262  }
263 
264  // Merge LHS/RHS.
265  LHS = new BinaryExprAST(BinOp, LHS, RHS);
266  }
267 }

285 PrototypeAST *Parser::ParsePrototype() {
286 
287  std::string FnName;
288 
289  unsigned Kind = 0; // 0 = identifier, 1 = unary, 2 = binary.
290  unsigned BinaryPrecedence = 30;
291 
292  switch (Tok.Kind) {
293  default:
294  return ErrorP("Expected function name in prototype");
295  case tok::tok_identifier:
296  FnName = Tok.IdentifierStr;
297  Kind = 0;
298  GetNextToken();
299  break;
300  case tok::tok_unary:
301  GetNextToken();
302  if (!isascii(Tok.Kind))
303  return ErrorP("Expected unary operator");
304  FnName = "unary";
305  FnName += (char)Tok.Kind;
306  Kind = 1;
307  GetNextToken();
308  break;
309  case tok::tok_binary:
310  GetNextToken();
311  if (!isascii(Tok.Kind))
312  return ErrorP("Expected binary operator");
313  FnName = "binary";
314  FnName += (char)Tok.Kind;
315  Kind = 2;
316  GetNextToken();
317 
318  // Read the precedence if present.
319  if (Tok.Kind == tok::tok_number) {
320  if (Tok.NumVal < 1 || Tok.NumVal > 100)
321  return ErrorP("Invalid precedecnce: must be 1..100");
322  BinaryPrecedence = (unsigned)(Tok.NumVal);
323  GetNextToken();
324  }
325  break;
326  }
327 
328  if (Tok.Kind != '(')
329  return ErrorP("Expected '(' in prototype");
330 
331  std::vector<std::string> ArgNames;
332  while (GetNextToken() == tok::tok_identifier)
333  ArgNames.push_back(Tok.IdentifierStr);
334  if (Tok.Kind != ')')
335  return ErrorP("Expected ')' in prototype");
336 
337  // success.
338  GetNextToken(); // eat ')'.
339 
340  // Verify right number of names for operator.
341  if (Kind && ArgNames.size() != Kind)
342  return ErrorP("Invalid number of operands for operator");
343 
344  return new PrototypeAST(FnName, ArgNames, Kind != 0, BinaryPrecedence);
345 }
