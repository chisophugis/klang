73 llvm::Value *BinaryExprAST::Codegen() {
74  // Special case '=' because we don't want to emit the LHS as an expression.
75  if (Op == '=') {
76  // Assignment requires the LHS to be an identifier.
77  VariableExprAST *LHSE = llvm::dyn_cast<VariableExprAST>(LHS);
78  if (!LHSE)
79  return ErrorV("destination of '=' must be a variable");
80  // Codegen the RHS.
81  llvm::Value *Val = RHS->Codegen();
82  if (Val == 0) return 0;
83 
84  // Look up the name.
85  llvm::Value *Variable = NamedValues[LHSE->getName()];
86  if (Variable == 0) return ErrorV("Unknown variable name");
87 
88  Builder.CreateStore(Val, Variable);
89  return Val;
90  }
91 
92  llvm::Value *L = LHS->Codegen();
93  llvm::Value *R = RHS->Codegen();
94  if (L == 0 || R == 0) return 0;
95 
96  switch (Op) {
97  case '+': return Builder.CreateFAdd(L, R, "addtmp");
98  case '-': return Builder.CreateFSub(L, R, "subtmp");
99  case '*': return Builder.CreateFMul(L, R, "multmp");
100  case '<':
101  L = Builder.CreateFCmpULT(L, R, "cmptmp");
102  // Convert bool 0/1 to double 0.0 or 1.0
103  return Builder.CreateUIToFP(
104  L,
105  llvm::Type::getDoubleTy(llvm::getGlobalContext()),
106  "booltmp");
107  default: break;
108  }
109 
110  // If it wasn't a builtin binary operator, it must be a user defined one. Emit
111  // a call to it.
112  llvm::Function *F = TheModule->getFunction(std::string("binary")+Op);
113  assert(F && "binary operator not found!");
114 
115  llvm::Value *Ops[2] = { L, R };
116  return Builder.CreateCall(F, Ops, "binop");
117 }

119 llvm::Value *CallExprAST::Codegen() {
120  // Look up the name in the global module table.
121  llvm::Function *CalleeF = TheModule->getFunction(Callee);
122  if (CalleeF == 0)
123  return ErrorV("Unknown function referenced");
124 
125  // If argument mismatch error.
126  if (CalleeF->arg_size() != Args.size())
127  return ErrorV("Incorrect # arguments passed");
128 
129  std::vector<llvm::Value*> ArgsV;
130  for (unsigned i = 0, e = Args.size(); i != e; ++i) {
131  ArgsV.push_back(Args[i]->Codegen());
132  if (ArgsV.back() == 0) return 0;
133  }
134 
135  return Builder.CreateCall(CalleeF, ArgsV, "calltmp");
136 }

138 llvm::Value *IfExprAST::Codegen() {
139  llvm::Value *CondV = Cond->Codegen();
140  if (CondV == 0) return 0;
141 
142  // Convert condition to a bool by comparing equal to 0.0.
143  CondV = Builder.CreateFCmpONE(
144  CondV,
145  llvm::ConstantFP::get(llvm::getGlobalContext(), llvm::APFloat(0.0)),
146  "ifcond");
147 
148  llvm::Function *TheFunction = Builder.GetInsertBlock()->getParent();
149 
150  // Create blocks for the then and else cases. Insert the 'then' block at the
151  // end of the function.
152  llvm::BasicBlock *ThenBB = llvm::BasicBlock::Create(
153  llvm::getGlobalContext(),
154  "then",
155  TheFunction);
156  llvm::BasicBlock *ElseBB = llvm::BasicBlock::Create(
157  llvm::getGlobalContext(),
158  "else");
159  llvm::BasicBlock *MergeBB = llvm::BasicBlock::Create(
160  llvm::getGlobalContext(),
161  "ifcont");
162 
163  Builder.CreateCondBr(CondV, ThenBB, ElseBB);
164 
165  // Emit then value.
166  Builder.SetInsertPoint(ThenBB);
167 
168  llvm::Value *ThenV = Then->Codegen();
169  if (ThenV == 0) return 0;
170 
171  Builder.CreateBr(MergeBB);
172  // Codegen of 'Then' can change the current block, update ThenBB for the PHI.
173  ThenBB = Builder.GetInsertBlock();
174 
175  // Emit else block.
176  TheFunction->getBasicBlockList().push_back(ElseBB);
177  Builder.SetInsertPoint(ElseBB);
178 
179  llvm::Value *ElseV = Else->Codegen();
180  if (ElseV == 0) return 0;
181 
182  Builder.CreateBr(MergeBB);
183  // Codegen of 'Else' can change the current block, update ElseBB for the PHI.
184  ElseBB = Builder.GetInsertBlock();
185 
186  // Emit merge block.
187  TheFunction->getBasicBlockList().push_back(MergeBB);
188  Builder.SetInsertPoint(MergeBB);
189  llvm::PHINode *PN = Builder.CreatePHI(
190  llvm::Type::getDoubleTy(llvm::getGlobalContext()),
191  2,
192  "iftmp");
193 
194  PN->addIncoming(ThenV, ThenBB);
195  PN->addIncoming(ElseV, ElseBB);
196  return PN;
197 }

199 llvm::Value *ForExprAST::Codegen() {
200  // Output this as:
201  // var = alloca double
202  // ...
203  // start = startexpr
204  // store start -> var
205  // goto loop
206  // loop:
207  // ...
208  // bodyexpr
209  // ...
210  // loopend:
211  // step = stepexpr
212  // endcond = endexpr
213  //
214  // curvar = load var
215  // nextvar = curvar + step
216  // store nextvar -> var
217  // br endcond, loop, endloop
218  // outloop:
219 
220  llvm::Function *TheFunction = Builder.GetInsertBlock()->getParent();
221 
222  // Create an alloca for the variable in the entry block.
223  llvm::AllocaInst *Alloca = CreateEntryBlockAlloca(TheFunction, VarName);
224 
225  // Emit the start code first, without 'variable' in scope.
226  llvm::Value *StartVal = Start->Codegen();
227  if (StartVal == 0) return 0;
228 
229  // Store the value into the alloca.
230  Builder.CreateStore(StartVal, Alloca);
231 
232  // Make the new basic block for the loop header, inserting after current
233  // block.
234  llvm::BasicBlock *LoopBB = llvm::BasicBlock::Create(
235  llvm::getGlobalContext(),
236  "loop",
237  TheFunction);
238 
239  // Insert an explicit fall through from the current block to the LoopBB.
240  Builder.CreateBr(LoopBB);
241 
242  // Start insertion in LoopBB.
243  Builder.SetInsertPoint(LoopBB);
244 
245  // Within the loop, the variable is defined equal to the PHI node. If it
246  // shadows an existing variable, we have to restore it, so save it now.
247  llvm::AllocaInst *OldVal = NamedValues[VarName];
248  NamedValues[VarName] = Alloca;
249 
250  // Emit the body of the loop. This, like any other expr, can change the
251  // current BB. Note that we ignore the value computed by the body, but don't
252  // allow an error.
253  if (Body->Codegen() == 0)
254  return 0;
255 
256  // Emit the step value.
257  llvm::Value *StepVal;
258  if (Step) {
259  StepVal = Step->Codegen();
260  if (StepVal == 0) return 0;
261  } else {
262  // If not specified, use 1.0.
263  StepVal = llvm::ConstantFP::get(
264  llvm::getGlobalContext(),
265  llvm::APFloat(1.0));
266  }
267 
268  // Compute the end condition.
269  llvm::Value *EndCond = End->Codegen();
270  if (EndCond == 0) return EndCond;
271 
272  // Reload, increment, and restore the alloca. This handles the case where
273  // the body of the loop mutates the variable.
274  llvm::Value *CurVar = Builder.CreateLoad(Alloca, VarName.c_str());
275  llvm::Value *NextVar = Builder.CreateFAdd(CurVar, StepVal, "nextvar");
276  Builder.CreateStore(NextVar, Alloca);
277 
278  // Convert condition to a bool by comparing equal to 0.0.
279  EndCond = Builder.CreateFCmpONE(
280  EndCond,
281  llvm::ConstantFP::get(llvm::getGlobalContext(), llvm::APFloat(0.0)),
282  "loopcond");
283 
284  // Create the "after loop" block and insert it.
285  llvm::BasicBlock *AfterBB = llvm::BasicBlock::Create(
286  llvm::getGlobalContext(),
287  "afterloop",
288  TheFunction);
289 
290  // Insert the conditional branch into the end of LoopEndBB.
291  Builder.CreateCondBr(EndCond, LoopBB, AfterBB);
292 
293  // Any new code will be inserted in AfterBB.
294  Builder.SetInsertPoint(AfterBB);
295 
296  // Restore the unshadowed variable.
297  if (OldVal)
298  NamedValues[VarName] = OldVal;
299  else
300  NamedValues.erase(VarName);
301 
302  // for expr always returns 0.0.
303  return llvm::Constant::getNullValue(
304  llvm::Type::getDoubleTy(llvm::getGlobalContext()));
305 }

308 llvm::Value *VarExprAST::Codegen() {
309  std::vector<llvm::AllocaInst *> OldBindings;
310 
311  llvm::Function *TheFunction = Builder.GetInsertBlock()->getParent();
312 
313  // Register all variables and emit their initializer.
314  for (unsigned i = 0, e = VarNames.size(); i != e; ++i) {
315  const std::string &VarName = VarNames[i].first;
316  ExprAST *Init = VarNames[i].second;
317 
318  // Emit the initializer before adding the variable to scope, this prevents
319  // the initializer from referencing the variable itself, and permits stuff
320  // like this:
321  // var a = 1 in
322  // var a = a in ... # refers to outer 'a'.
323  llvm::Value *InitVal;
324  if (Init) {
325  InitVal = Init->Codegen();
326  if (InitVal == 0) return 0;
327  } else { // If not specified, use 0.0.
328  InitVal = llvm::ConstantFP::get(llvm::getGlobalContext(),
329  llvm::APFloat(0.0));
330  }
331 
332  llvm::AllocaInst *Alloca = CreateEntryBlockAlloca(TheFunction, VarName);
333  Builder.CreateStore(InitVal, Alloca);
334 
335  // Remember the old variable binding so that we can restore the binding when
336  // we unrecurse.
337  OldBindings.push_back(NamedValues[VarName]);
338 
339  // Remember this binding.
340  NamedValues[VarName] = Alloca;
341  }
342 
343  // Codegen the body, now that all vars are in scope.
344  llvm::Value *BodyVal = Body->Codegen();
345  if (BodyVal == 0) return 0;
346 
347  // Pop all our variables from scope.
348  for (unsigned i = 0, e = VarNames.size(); i != e; ++i)
349  NamedValues[VarNames[i].first] = OldBindings[i];
350 
351  // Return the body computation.
352  return BodyVal;
353 }

356 llvm::Function *PrototypeAST::Codegen() {
357  // Make the function type: double(double,double) etc.
358  std::vector<llvm::Type*> Doubles(
359  Args.size(),
360  llvm::Type::getDoubleTy(llvm::getGlobalContext()));
361  llvm::FunctionType *FT = llvm::FunctionType::get(
362  llvm::Type::getDoubleTy(llvm::getGlobalContext()),
363  Doubles,
364  false);
365 
366  llvm::Function *F = llvm::Function::Create(
367  FT,
368  llvm::Function::ExternalLinkage,
369  Name,
370  TheModule);
371 
372  // If F conflicted, there was already something named 'Name'. If it has a
373  // body, don't allow redefinition or reextern.
374  if (F->getName() != Name) {
375  // Delete the one we just made and get the existing one.
376  F->eraseFromParent();
377  F = TheModule->getFunction(Name);
378 
379  // If F already has a body, reject this.
380  if (!F->empty()) {
381  ErrorF("redefinition of function");
382  return 0;
383  }
384 
385  // If F took a different number of args, reject.
386  if (F->arg_size() != Args.size()) {
387  ErrorF("redefinition of function with different # args");
388  return 0;
389  }
390  }
391 
392  // Set names for all arguments.
393  unsigned Idx = 0;
394  for (llvm::Function::arg_iterator AI = F->arg_begin(); Idx != Args.size();
395  ++AI, ++Idx)
396  AI->setName(Args[Idx]);
397 
398  return F;
399 }

404 void PrototypeAST::CreateArgumentAllocas(llvm::Function *F) {
405  llvm::Function::arg_iterator AI = F->arg_begin();
406  for (unsigned Idx = 0, e = Args.size(); Idx != e; ++Idx, ++AI) {
407  // Create an alloca for this variable.
408  llvm::AllocaInst *Alloca = CreateEntryBlockAlloca(F, Args[Idx]);
409 
410  // Store the initial value into the alloca.
411  Builder.CreateStore(AI, Alloca);
412 
413  // Add arguments to variable symbol table.
414  NamedValues[Args[Idx]] = Alloca;
415  }

419 llvm::Function *FunctionAST::Codegen() {
420  NamedValues.clear();
421 
422  llvm::Function *TheFunction = Proto->Codegen();
423  if (TheFunction == 0)
424  return 0;
425 
426  // If this is an operator, install it.
427  if (Proto->isBinaryOp())
428  Token::BinopPrecedence[Proto->getOperatorName()] =
429  Proto->getBinaryPrecedence();
430 
431  // Create a new basic block to start insertion into.
432  llvm::BasicBlock *BB = llvm::BasicBlock::Create(
433  llvm::getGlobalContext(),
434  "entry",
435  TheFunction);
436  Builder.SetInsertPoint(BB);
437 
438  // Add all arguments to the symbol table and create their allocas.
439  Proto->CreateArgumentAllocas(TheFunction);
440 
441  if (llvm::Value *RetVal = Body->Codegen()) {
442  // Finish off the function.
443  Builder.CreateRet(RetVal);
444 
445  // Validate the generated code, checking for consistency.
446  llvm::verifyFunction(*TheFunction);
447 
448  //----------------------
449  // Optimize the function.
450  //----------------------
451  TheFPM->run(*TheFunction);
452 
453  return TheFunction;
454  }
455 
456  // Error reading body, remove function.
457  TheFunction->eraseFromParent();
458 
459  if (Proto->isBinaryOp())
460  Token::BinopPrecedence.erase(Proto->getOperatorName());
461  return 0;
462 }
